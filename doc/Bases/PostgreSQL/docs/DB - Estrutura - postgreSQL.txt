Schema de Dados no PostgreSQL para o ASSJUR

CREATE TABLE users (
    user_id SERIAL PRIMARY KEY,
    userrole VARCHAR(10) NOT NULL,
    username VARCHAR(20) NOT NULL,
    password VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (email),
    UNIQUE (username)
);

CREATE TABLE sessions (
    session_id SERIAL PRIMARY KEY,
    user_id INT,
    model VARCHAR(30) NOT NULL,
    prompt_tokens INT,
    completion_tokens INT,
    total_tokens INT,
    session_start TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    session_end TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE SET NULL
);

CREATE TABLE IF NOT EXISTS public.contexto
(
    id_ctxt integer NOT NULL DEFAULT nextval('contexto_id_ctxt_seq'::regclass),
    nr_proc character varying(24) COLLATE pg_catalog."default" NOT NULL,
    juizo character varying(255) COLLATE pg_catalog."default" NOT NULL,
    classe character varying(255) COLLATE pg_catalog."default" NOT NULL,
    assunto character varying(255) COLLATE pg_catalog."default" NOT NULL,
    prompt_tokens integer,
    completion_tokens integer,
    dt_inc date NOT NULL,
    status character(1) COLLATE pg_catalog."default" NOT NULL DEFAULT 'S'::bpchar,
    CONSTRAINT contexto_pkey PRIMARY KEY (id_ctxt)
)

CREATE TABLE IF NOT EXISTS public.uploads
(
    id_file integer NOT NULL DEFAULT nextval('temp_uploadfiles_id_file_seq'::regclass),
    id_ctxt integer NOT NULL,
    nm_file_new character varying(255) COLLATE pg_catalog."default",
    nm_file_ori character varying(255) COLLATE pg_catalog."default",
    sn_autos character(1) COLLATE pg_catalog."default" NOT NULL DEFAULT 'N'::bpchar,
    dt_inc date NOT NULL,
    status character(1) COLLATE pg_catalog."default" NOT NULL DEFAULT 'S'::bpchar,
    CONSTRAINT temp_uploadfiles_pkey PRIMARY KEY (id_file),
    CONSTRAINT temp_uploadfiles_id_ctxt_fkey FOREIGN KEY (id_ctxt)
        REFERENCES public.contexto (id_ctxt) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)

CREATE TABLE IF NOT EXISTS public.prompts
(
    id_prompt SERIAL PRIMARY KEY,
    id_nat integer NOT NULL,
    id_doc integer NOT NULL,
    id_classe integer NOT NULL,
    id_assunto integer NOT NULL,  
    nm_desc character varying(255) COLLATE pg_catalog."default",
    txt_prompt text COLLATE pg_catalog."default",
    dt_inc date NOT NULL,
    status character(1) COLLATE pg_catalog."default" NOT NULL DEFAULT 'S'::bpchar,
    )

Notas sobre a Conversão:
AUTO_INCREMENT: Substituído por SERIAL, que é uma forma comum de criar colunas de incremento automático no PostgreSQL.
ENGINE: A cláusula ENGINE=InnoDB foi removida, pois o PostgreSQL não requer essa especificação.
Chaves Únicas: A definição de chaves únicas foi ajustada para a sintaxe do PostgreSQL.
Charset e Collation: As definições de DEFAULT CHARSET e COLLATE foram removidas, pois o PostgreSQL lida com isso de maneira diferente e geralmente usa UTF-8 por padrão.

Explicação dos Valores:
userrole: O papel do usuário, neste caso, definido como 'admin'.
username: O nome de usuário, definido como 'johndoe'.
password: A senha do usuário, que neste exemplo é 'securepassword123'. Nota: Em um ambiente real, você deve sempre armazenar senhas de forma segura, utilizando hashing (por exemplo, bcrypt).
email: O endereço de e-mail do usuário, definido como 'johndoe@example.com'.
created_at: A data e hora de criação do registro, que é definida como CURRENT_TIMESTAMP, o que insere a data e hora atuais.
